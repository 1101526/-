int m_judge(int *m,int n)                                  //蓝量判断,同时减去消耗的蓝量
{
    int result;
    if(n<4)
        *m-=n;
    if(n>3&&n<7)
        *m-=(n-3);
    if(n>6&&n<10)
        *m-=(n-6);
    if(n>9&&n<13)
        *m-=(n-9);
    if(*m>=0)
        result=1;
    if(*m<0)
    result=0;
    return result;
}
