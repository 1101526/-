int m_judge(int *m,int n)                                  //蓝量判断,同时减去消耗的蓝量
{
    int result;
    if(n%4==1)
    *m-=1;
    if(n%4==2)
    *m-=2;
    if(n%4==3)
    *m-=3;
    if(*m>0)
    result=1;
    if(*m<0)
    result=0;
    return result;
}
